114 template<typename T, typename tolT>
115 struct results_rkab<T> *rkab(const int astages, const int bstages,
116  const T *ba, const T *bb, const T *a, const T *c,
117  T *u_init, int dim, int maxsteps, tolT tol,
118  T t, T t_end, void (*get_f)(T, T*, T*))
119 {
120  assert(astages < bstages);
121  // Set some resonable acceptance patterns
122  const T acc_scale = pow(0.9, bstages);
123  const T max_adapt = 10;
124  const T min_adapt = 0.5;
125 
126  // Initialize dynamically sized memory for holding results
127  vector<T> tvec;
128  vector<T> u;
129  // Allocate temporary arrays. On heap for safety; don't forget to delete!
130  T *ua = new T[dim];
131  T *ub = new T[dim];
132  T *u_k = new T[dim];
133  T *u_prev = new T[dim]; // for easy re-init on failure
134  T *f = new T[bstages * dim];
135  T *haf = new T[dim * (bstages - 1)];
136 
137  // Initialize
138  int numfailures = 0;
139  int numsteps = 0;
140  copy(u_init, u_init + dim, ua);
141  copy(u_init, u_init + dim, ub);
142  copy(u_init, u_init + dim, u_k);
143  copy(u_init, u_init + dim, u_prev);
144 
145  // 'a' is a flattened sparse rep of a triangular matrix; I'll iterate
146  // through its elements by advancing this pointer.
147  const T *a_k = a;
148 
149  // Guess an initial step size
150  T h = min(abs(t_end - t)/10, (T)0.1);
151  // Choose the sign of h according to the direction of propagation
152  int t_dir = (t_end >= t) ? 1 : -1;
153  h *= t_dir;
154 
155  // Main loop
156  while (numsteps < maxsteps && t_dir * (t_end - t) > 0)
157  {
158  bool failures = false;
159  T hmin = 16 * boost::math::ulp(t);
160  // hmin is the minimum meaningful magnitude of h
161  if (abs(h) < hmin) { // abs(h) should be >= hmin
162  h = t_dir * hmin;
163  }
164  // But make sure to hit the last step exactly
165  if (t_dir * (t_end - t - h) < 0){
166  h = t_end - t;
167  }
168 
169  // Loop for advancing one step
170  while (true)
171  {
172  // Please study these lines carefully. How do they work?
173  // (My tableau has leading zeroes dropped with 'a' transposed)
174  fill(haf, haf + (bstages - 1) * dim, 0); // reset temp 'haf' array
175  a_k = a; // reset to row k = 0
176  get_f(t, u_k, f); // stage 1
177  for (int k = 0; k < bstages - 1; ++k) // stages 2 through last
178  {
179  for (int i = 0; i < dim; ++i){
180  T hf_ki = h * f[k * dim + i];
181  // Update ua[i] and ub[i] now since hf_ki computed
182  ub[i] += bb[k] * hf_ki;
183  if (k < astages){ // Likely always true and optimized out
184  ua[i] += ba[k] * hf_ki;
185  }
186  for (int j = k; j < bstages - 1; ++j){
187  haf[i*(bstages - 1) + j] += hf_ki * a_k[j - k];
188  }
189  u_k[i] = u_prev[i] + haf[i*(bstages - 1) + k];
190  }
191  get_f(t + h * c[k], u_k, &f[(k + 1) * dim]);
192  a_k += bstages - 1 - k; // advance to next row
193  }
194  for (int i = 0; i < dim; ++i){ // finish last stage
195  ub[i] += h * bb[bstages - 1] * f[(bstages - 1) * dim + i];
196  } // Note: I asserted astages < bstages, so ua is already complete.
197 
198  // acceptability = (tolerance) / (relative error)
199  T acceptability = acc_scale * acceptability_rel(dim, ua, ub, tol);
200  // If the step is acceptable or the step size is minimal
201  if (acceptability > 1 || abs(h) <= hmin)
202  { // Accept the step
203  ++numsteps;
204  t += h;
205  tvec.push_back(t);
206  copy(ub, ub + dim, ua);
207  copy(ub, ub + dim, u_prev);
208  copy(ub, ub + dim, u_k);
209  u.insert(u.end(), ub, ub + dim);
210  // Adapt step size; don't increase by a factor > max_adapt
211  h *= min(max_adapt, (T)(pow(acceptability, 1.0/bstages)));
212  break;
213  }
214  else
215  { // Reject the step
216  if (!failures)
217  {
218  failures = true;
219  ++numfailures;
220  // Adapt step size; don't decrease by a factor < min_adapt
221  h *= max(min_adapt, (T)(pow(acceptability, 1.0/bstages)));
222  } else { // We underestimated error! Be pessimistic.
223  h *= min_adapt;
224  }
225  copy(u_prev, u_prev + dim, ua);
226  copy(u_prev, u_prev + dim, ub);
227  copy(u_prev, u_prev + dim, u_k);
228  }
229  }
230  }
231  assert(tvec.size() == (size_t)numsteps);
232  assert(u.size() == (size_t)numsteps * dim);
233 
234  // Allocate memory dynamically (persists outside function) for results
235  T *tarr = new T[numsteps]; // time array
236  T *uarr = new T[numsteps * dim]; // u array
237  results_rkab<T> *results = new results_rkab<T> {
238  numsteps,
239  tarr, // pointer
240  uarr, // pointer
241  numfailures
242  };
243  // Copy time and u data from the vectors.
244  // Can't return vectors directly because I want a C-compatible interface.
245  copy(tvec.begin(), tvec.end(), tarr);
246  copy(u.begin(), u.end(), uarr);
247 
248  // delete temporary arrays
249  delete [] ua;
250  delete [] ub;
251  delete [] u_k;
252  delete [] u_prev;
253  delete [] f;
254  delete [] haf;
255  // vectors are RAII; they'll be deleted automatically.
256 
257  return results;
258 }
